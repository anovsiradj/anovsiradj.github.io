HTML Template:
<template id="shadow-button-template">
  <style>
    /* Styles are encapsulated in Shadow DOM */
    button {
      padding: 12px 24px;
      font-size: 16px;
      border: none;
      border-radius: 6px;
      cursor: pointer;
      color: white;
      font-weight: bold;
      transition: transform 0.1s;
    }
    button:hover {
      transform: scale(1.05);
    }
  </style>
  <button></button>
</template>
JavaScript:
component('shadow-button', '#shadow-button-template', {
  shadow: true, // Enable Shadow DOM
  attrs: ['label', 'color'],
  
  mounted(root) {
    this.updateButton();
  },
  
  changed(attr) {
    if (attr === 'label' || attr === 'color') {
      this.updateButton();
    }
  },
  
  on: {
    'button': {
      click() {
        this.dispatchEvent(new CustomEvent('shadow-click', {
          bubbles: true,
          detail: { label: this.getAttribute('label') }
        }));
      }
    }
  },
  
  methods: {
    updateButton() {
      const btn = this.shadowRoot.querySelector('button');
      btn.textContent = this.getAttribute('label') || 'Click';
      btn.style.background = this.getAttribute('color') || '#3498db';
    }
  }
});
Shadow DOM Benefits
  • Styles are completely isolated from the rest of the page
  • No CSS conflicts with global styles
  • Component internals are encapsulated
  • Perfect for reusable UI components
