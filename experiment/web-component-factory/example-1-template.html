JavaScript:
component('greeting-card', '#greeting-template', {
  attrs: ['name', 'message'],
  
  mounted() {
    this.updateGreeting();
  },
  
  changed(attr, oldVal, newVal) {
    if (attr === 'name' || attr === 'message') {
      this.updateGreeting();
    }
  },
  
  on: {
    '.greet-btn': {
      click() {
        const name = this.getAttribute('name') || 'Guest';
        this.dispatchEvent(new CustomEvent('greeting-sent', {
          bubbles: true,
          detail: { name, timestamp: Date.now() }
        }));
      }
    }
  },
  
  methods: {
    updateGreeting() {
      const name = this.getAttribute('name') || 'Guest';
      const message = this.getAttribute('message') || 'Welcome!';
      this.querySelector('.greeting-text').textContent = \`Hello, \${name}!\`;
      this.querySelector('.message-text').textContent = message;
    }
  }
});
