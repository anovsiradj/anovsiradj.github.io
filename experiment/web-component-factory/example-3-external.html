JavaScript:
component('user-card', './user-card.html', {
  attrs: ['user-name', 'email', 'role'],
  
  mounted() {
    this.updateUserInfo();
  },
  
  changed(attr) {
    this.updateUserInfo();
  },
  
  on: {
    '.edit': {
      click() {
        this.dispatchEvent(new CustomEvent('user-edit', {
          bubbles: true,
          detail: { 
            name: this.getAttribute('user-name'),
            email: this.getAttribute('email')
          }
        }));
      }
    },
    '.delete': {
      click() {
        const userName = this.getAttribute('user-name');
        if (confirm(\`Delete user \${userName}?\`)) {
          this.dispatchEvent(new CustomEvent('user-delete', {
            bubbles: true,
            detail: { name: userName }
          }));
          this.remove();
        }
      }
    }
  },
  
  methods: {
    updateUserInfo() {
      this.querySelector('.name').textContent = this.getAttribute('user-name');
      this.querySelector('.email').textContent = \`Email: \${this.getAttribute('email')}\`;
      this.querySelector('.role').textContent = \`Role: \${this.getAttribute('role')}\`;
    }
  }
});
Important Note

External HTML files require a local server to work (due to CORS). Use Python's python3 -m http.server or similar.
