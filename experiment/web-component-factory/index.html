Quick Start

		import { component } from './component-factory.js';

component('my-button', '#button-template', {
  attrs: ['label'],
  mounted() {
    console.log('Button ready!');
  },
  on: {
    'button': {
      click() {
        alert('Clicked!');
      }
    }
  }
});
